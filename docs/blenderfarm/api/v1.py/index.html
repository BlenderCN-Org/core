        }

    def clear_local_state(self):

Clears local state. Used in case of aborted connections or when disconnecting.

High-level connect/disconnect

    def connect(self, user, key):

Connects to the server. Raises an Error if connection is unsuccessful.

        self.user = user
        self.key = key

        try:
            self.request_server_info()
            self.request_auth_test()
        except error.Error as exception:
            self.clear_local_state()
            
            raise exception # pylint: disable=raising-bad-type

        self.connection_start_time = time.monotonic()

        self.connected = True

    def disconnect(self):
